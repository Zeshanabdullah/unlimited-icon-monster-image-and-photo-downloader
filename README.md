### Python Example
This example demonstrates how to download an image using the `requests` library in Python. It handles simple errors such as connection issues.

python
import requests

def download_image(image_url, save_path):
    try:
        response = requests.get(image_url)
        response.raise_for_status()  # Check for HTTP errors
        with open(save_path, 'wb') as file:
            file.write(response.content)
        print(f"Image successfully downloaded: {save_path}")
    except requests.exceptions.RequestException as e:
        print(f"Error downloading image: {e}")

### PHP Example
In this PHP example, we use cURL to fetch and save an image from the specified URL. Error handling is included to capture any issues during the request.

php
<?php
function downloadImage($url, $savePath) {
    $ch = curl_init($url);
    $fp = fopen($savePath, 'wb');

    curl_setopt($ch, CURLOPT_FILE, $fp);
    curl_setopt($ch, CURLOPT_HEADER, 0);
    
    if (!curl_exec($ch)) {
        echo 'Error downloading image: ' . curl_error($ch);
    } else {
        echo "Image successfully downloaded: $savePath\n";
    }

    curl_close($ch);
    fclose($fp);
}

// Example usage
$imageUrl = 'https://hdstockimages.com/ancestory-image-and-photo-downloader/image.jpg';
downloadImage($imageUrl, 'downloaded_image.jpg');
?>

async function downloadImage(imageUrl, savePath) {
    try {
        const response = await fetch(imageUrl);
        if (!response.ok) throw new Error('Network response was not ok');

        const blob = await response.blob();
        const link = document.createElement('a');
        link.href = URL.createObjectURL(blob);
        link.download = savePath;
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        console.log(`Image successfully downloaded: ${savePath}`);
    } catch (error) {
        console.error('Error downloading image:', error);
    }
}
